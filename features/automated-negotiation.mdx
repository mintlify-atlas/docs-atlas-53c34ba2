## Overview

Haggle's most powerful feature is its **AI voice agent** that automatically calls service providers, explains the job, and negotiates the best price on your behalf. Powered by **Grok Realtime API** and **Twilio**, the agent conducts natural phone conversations to secure competitive quotes.

<Warning>
  The voice agent makes real phone calls to actual service providers. Ensure you have valid Twilio credentials and XAI API access configured.
</Warning>

## How It Works

<Steps>
  <Step title="User Initiates Negotiation">
    After providers are found, user clicks "Start Negotiating" in the UI
  </Step>
  
  <Step title="Voice Agent Calls Providers">
    Twilio initiates simultaneous calls to all providers for the job
  </Step>
  
  <Step title="Real-Time Conversation">
    Grok Realtime API conducts natural voice conversations, explaining the job and negotiating price
  </Step>
  
  <Step title="Transcript Captured">
    All audio is transcribed in real-time for analysis
  </Step>
  
  <Step title="Price Extracted">
    Grok LLM analyzes the transcript to extract the final negotiated price
  </Step>
  
  <Step title="Results Displayed">
    User sees negotiated prices from all providers and selects the best offer
  </Step>
</Steps>

```python main.py:363-402
@app.post("/api/start-calls/{job_id}")
async def start_calls(job_id: str):
    """
    Trigger calls for all providers in a job.
    This calls the backend/app.py service running on port 6000.
    """
    # Verify job exists
    job = jobs_store.get(job_id)
    if not job:
        raise HTTPException(status_code=404, detail=f"Job not found: {job_id}")
    
    # Verify providers exist
    providers = get_providers_by_job_id(job_id)
    if not providers:
        raise HTTPException(status_code=404, detail=f"No providers found for job: {job_id}")
    
    # Call the backend service to start calls
    backend_url = os.getenv("CALL_BACKEND_URL", "http://localhost:6000")
    
    try:
        async with httpx.AsyncClient(timeout=30.0) as client:
            response = await client.post(f"{backend_url}/start-job/{job_id}")
            if response.status_code == 200:
                result = response.json()
                return {
                    "status": "started",
                    "message": f"Started calls for {result.get('count', len(providers))} providers",
                    "provider_count": result.get("count", len(providers))
                }
            else:
                raise HTTPException(
                    status_code=response.status_code,
                    detail=f"Failed to start calls: {response.text}"
                )
    except httpx.RequestError as e:
        raise HTTPException(
            status_code=503,
            detail=f"Call backend service unavailable: {str(e)}"
        )
```

### WebSocket Audio Streaming

The core of the system is a WebSocket that streams audio between Twilio and Grok:

```python app.py:105-224
@app.websocket("/media-stream")
async def handle_media_stream(websocket: WebSocket):
    await websocket.accept()
    
    transcript = []
    provider_id = None
    provider = None

    try:
        async with websockets.connect(GROK_URL, additional_headers={"Authorization": f"Bearer {API_KEY}"}) as grok_ws:
            
            stream_sid = None
            
            async def receive_from_twilio():
                nonlocal stream_sid, provider, provider_id
                try:
                    while True:
                        msg = await websocket.receive_text()
                        data = json.loads(msg)
                        
                        if data['event'] == 'start':
                            stream_sid = data['start']['streamSid']
                            custom_params = data['start']['customParameters']
                            provider_id = custom_params.get('provider_id')
                            
                            # Update call status to in_progress
                            if provider_id:
                                try:
                                    update_provider_call_status(int(provider_id), "in_progress")
                                except Exception as e:
                                    print(f"⚠️  Failed to update call status: {e}")
                            
                            # Fetch provider details from database
                            data = supabase.table("providers").select("*").eq("id", provider_id).single().execute()
                            provider = data.data
                            print(f"🔌 Connected to provider: {provider.get('service_provider')}")
                            
                            # Configure the Grok session
                            await grok_ws.send(json.dumps({
                                "type": "session.update",
                                "session": {
                                    "voice": "Rex",
                                    "instructions": generate_prompt(provider),
                                    "turn_detection": {"type": "server_vad"},
                                    "audio": {
                                        "input": {"format": {"type": "audio/pcm", "rate": 24000}},
                                        "output": {"format": {"type": "audio/pcm", "rate": 24000}}
                                    }
                                }
                            }))
                            
                            # Trigger the greeting
                            await grok_ws.send(json.dumps({
                                "type": "response.create"
                            }))
                            
                        elif data['event'] == 'media':
                            # Convert mulaw audio from Twilio to PCM for Grok
                            mulaw = base64.b64decode(data['media']['payload'])
                            pcm_8k = audioop.ulaw2lin(mulaw, 2)
                            pcm_24k, _ = audioop.ratecv(pcm_8k, 2, 1, 8000, 24000, None)
                            
                            await grok_ws.send(json.dumps({
                                "type": "input_audio_buffer.append",
                                "audio": base64.b64encode(pcm_24k).decode('utf-8')
                            }))
                            
                except WebSocketDisconnect:
                    raise
                except Exception as e:
                    pass

            async def send_to_twilio():
                nonlocal transcript
                try:
                    async for msg in grok_ws:
                        event = json.loads(msg)
                        event_type = event.get('type')
                        
                        if event_type== 'response.output_audio.delta':
                            # Convert PCM audio from Grok to mulaw for Twilio
                            pcm_24k = base64.b64decode(event['delta'])
                            pcm_8k, _ = audioop.ratecv(pcm_24k, 2, 1, 24000, 8000, None)
                            mulaw = audioop.lin2ulaw(pcm_8k, 2)
                            
                            if stream_sid:
                                await websocket.send_json({
                                    "event": "media",
                                    "streamSid": stream_sid,
                                    "media": {"payload": base64.b64encode(mulaw).decode('utf-8')}
                                })

                        # Capture transcript
                        elif event_type == 'conversation.item.input_audio_transcription.completed':
                            user_text = event.get('transcript', '')
                            if user_text:
                                transcript.append({"role": "user", "text": user_text})
                                print(f"[USER]: {user_text}")
                        
                        elif event_type == 'response.audio_transcript.done':
                            asst_text = event.get('transcript', '')
                            if asst_text:
                                transcript.append({"role": "assistant", "text": asst_text})
                                print(f"[ASSISTANT]: {asst_text}")

                except Exception as e:
                    pass

            # Run both audio loops concurrently
            await asyncio.gather(receive_from_twilio(), send_to_twilio())
```

<Note>
  Audio conversion is critical: Twilio uses 8kHz mulaw format, while Grok Realtime uses 24kHz PCM. The system converts between formats in real-time using the `audioop` library.
</Note>

### Price Extraction with LLM

Grok LLM analyzes the transcript to extract the final negotiated price:

```python grok_llm.py:362-447
async def extract_negotiated_price(transcript: List[Dict[str, str]]) -> Optional[float]:
    """
    Extract the negotiated price from a call transcript using Grok LLM.
    
    Args:
        transcript: List of transcript entries with 'role' and 'text' keys
        
    Returns:
        Negotiated price as float, or None if no price was agreed upon
    """
    if not transcript:
        return None
    
    # Format transcript for LLM
    transcript_text = "\n".join([
        f"[{entry['role'].upper()}]: {entry['text']}"
        for entry in transcript
    ])
    
    system_prompt = """You are analyzing a phone call transcript between a homeowner and a service provider.
Your task is to extract the FINAL AGREED-UPON PRICE that was negotiated during the call.

IMPORTANT RULES:
1. Look for the final price that was agreed upon, not initial quotes
2. The price should be a number (e.g., 125, 150.50, 200)
3. If no price was agreed upon, respond with "none"
4. If multiple prices are mentioned, use the FINAL agreed price
5. Respond with ONLY the numeric value, nothing else (e.g., "125" or "150.50")
6. If the call ended without agreement, respond with "none"

Examples:
- "$125" -> "125"
- "one hundred twenty five dollars" -> "125"
- "We agreed on $150" -> "150"
- "I'll do it for $200" -> "200"
- No agreement reached -> "none" """

    user_prompt = f"""Call transcript:
{transcript_text}

What was the final agreed-upon price? Respond with only the number or "none" if no price was agreed."""

    try:
        client = Client(api_key=XAI_API_KEY)
        chat = client.chat.create(model="grok-3-fast")
        
        chat.append(system(system_prompt))
        chat.append(user(user_prompt))
        
        # Get response
        full_response = ""
        for response, chunk in chat.stream():
            if chunk.content:
                full_response += chunk.content
        
        price_str = full_response.strip().lower()
        
        # Handle "none" or empty responses
        if not price_str or price_str == "none" or "no" in price_str or "not" in price_str:
            return None
        
        # Extract numeric value
        import re
        numbers = re.findall(r'\d+\.?\d*', price_str)
        if numbers:
            try:
                price = float(numbers[0])
                return price
            except ValueError:
                pass
        
        return None
        
    except Exception as e:
        print(f"Grok API exception during price extraction: {e}")
        return _fallback_extract_price(transcript)
```

## Negotiation Strategies

The AI agent employs several human-like negotiation tactics:

<AccordionGroup>
  <Accordion title="Price Anchoring">
    The agent mentions the user's budget range early to set expectations:
    
    > "My budget is around $150" or "I was hoping to stay under $200"
  </Accordion>
  
  <Accordion title="Competitive Pressure">
    Implies comparison shopping without being aggressive:
    
    > "I'm getting a few quotes" or "That seems a bit high compared to what I've heard"
  </Accordion>
  
  <Accordion title="Flexibility Incentives">
    Offers scheduling flexibility for better pricing:
    
    > "I'm flexible on timing if that helps with the price"
  </Accordion>
  
  <Accordion title="Polite Persistence">
    Continues negotiating while remaining respectful:
    
    > "I appreciate that, but is there any way you could work with me on price?"
  </Accordion>
  
  <Accordion title="Graceful Exit">
    Ends calls professionally whether agreement is reached or not:
    
    > **Agreement:** "Thank you for your help! I'll reach out to you shortly."
    > 
    > **No Agreement:** "Thank you for the info. I need to think about it and will call you back."
  </Accordion>
</AccordionGroup>

## Monitoring Call Progress

The frontend can poll for real-time updates during calls:

```javascript Frontend Polling
const monitorCalls = (jobId) => {
  const pollInterval = setInterval(async () => {
    const response = await fetch(`/api/providers/${jobId}/status`);
    const providers = await response.json();
    
    providers.forEach(provider => {
      console.log(`${provider.name}:`);
      console.log(`  Status: ${provider.call_status}`);
      
      if (provider.negotiated_price) {
        console.log(`  Price: $${provider.negotiated_price}`);
      }
      
      if (provider.call_transcript) {
        console.log(`  Transcript available`);
      }
    });
    
    // Check if all calls are done
    const allComplete = providers.every(p => 
      ['completed', 'failed'].includes(p.call_status)
    );
    
    if (allComplete) {
      clearInterval(pollInterval);
      console.log('All calls completed!');
      
      // Find best price
      const successful = providers.filter(p => p.negotiated_price);
      const bestOffer = successful.reduce((min, p) => 
        p.negotiated_price < min.negotiated_price ? p : min
      );
      
      console.log(`Best offer: $${bestOffer.negotiated_price} from ${bestOffer.name}`);
    }
  }, 3000);  // Poll every 3 seconds
};
```

```python Database Update
def update_provider_call_status(
    provider_id: int,
    status: str,
    negotiated_price: Optional[float] = None,
    call_transcript: Optional[str] = None
):
    """Update provider record after call completes."""
```
